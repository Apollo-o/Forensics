import requests

url = "https://github.com/Apollo-o/Whistle/blob/master/Tools/Temp.txt"

try:
  response = requests.get(url)
  for header in response.headers: print('%s: %s' % (header, response.headers[header]))
except: print('%s: Not found' % header)
